700 template<typename T>
701 std::istream& operator>>(std::istream& str, enumRefHolder<T> const& data){
702  std::string value;
703  str >> value;
704 
705  // These two can be made easier to read in C++11
706  // using std::begin() and std::end()
707  //
708  static auto begin = std::begin(enumStrings<T>::data);
709  static auto end = std::end(enumStrings<T>::data);
710 
711  auto find = std::find(begin, end, value);
712  if (find != end){
713  data.enumVal = static_cast<T>(std::distance(begin, find));
714  }
715  return str;
716 }
